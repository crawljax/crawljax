CrawljaxConfiguration crawljax = new CrawljaxConfiguration();

InputSpecification input = new InputSpecification();
input.field("male").setValue(true);
input.field("name").setValue("Bob");
input.fields("phone", "mobile").setValue("1234567890");
input.field("type").setValue("Student");
input.field("active").setValue(true);
...
crawljax.setInputSpecification(input);
